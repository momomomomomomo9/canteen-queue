<body>


    <div class="card" id="student-view">

        <h2>学生取号</h2>

        <div id="my-ticket-box" style="display:none;">

            <p>您的号码是：</p>

            <div class="big-num" id="my-number">---</div>

            <p class="info" id="wait-info">前面还有 0 人在等待</p>

            <button onclick="cancelQueue()" style="background:#e74c3c">取消排队</button>

        </div>

        <div id="get-ticket-box">

            <p>目前排队总数：<span id="total-waiting">0</span> 人</p>

            <button onclick="getTicket()">立即取号</button>

        </div>

    </div>


    <div class="card" id="admin-view">

        <h2>食堂叫号管理</h2>

        <div class="info">当前正在处理：</div>

        <div class="big-num" id="current-calling">无</div>

        <button onclick="nextNumber()" style="background:#3498db">呼叫下一位</button>

        <button onclick="resetQueue()" style="background:#95a5a6">营业结束/清零</button>

        <ul class="call-list" id="queue-list"></ul>

    </div>


    <script>

        // 模拟数据库数据

        let queue = []; // 等待队列

        let currentNum = 0; // 当前叫号

        let myTicket = null; // 我的号码


        // 初始化显示

        updateDisplay();


        // --- 学生逻辑 ---

        function getTicket() {

            let nextNum = (queue.length > 0 ? queue[queue.length - 1] : (currentNum || 0)) + 1;

            queue.push(nextNum);

            myTicket = nextNum;

            showStudentStatus();

            updateDisplay();

        }


        function showStudentStatus() {

            document.getElementById('get-ticket-box').style.display = 'none';

            document.getElementById('my-ticket-box').style.display = 'block';

            document.getElementById('my-number').innerText = myTicket.toString().padStart(3, '0');

        }


        function cancelQueue() {

            queue = queue.filter(n => n !== myTicket);

            myTicket = null;

            document.getElementById('get-ticket-box').style.display = 'block';

            document.getElementById('my-ticket-box').style.display = 'none';

            updateDisplay();

        }


        // --- 管理员逻辑 ---

        function nextNumber() {

            if (queue.length > 0) {

                currentNum = queue.shift();

                // 语音播报功能

                speak(`请 ${currentNum} 号同学到档口取餐`);

                if(myTicket === currentNum) {

                    alert("到你了！请快去取餐！");

                    myTicket = null;

                    document.getElementById('get-ticket-box').style.display = 'block';

                    document.getElementById('my-ticket-box').style.display = 'none';

                }

                updateDisplay();

            } else {

                alert("目前没有人排队哦");

            }

        }


        function resetQueue() {

            if(confirm("确定要全部清零吗？")) {

                queue = [];

                currentNum = 0;

                myTicket = null;

                updateDisplay();

                location.reload(); 

            }

        }


        // --- 通用更新界面 ---

        function updateDisplay() {

            // 更新管理端显示

            document.getElementById('current-calling').innerText = currentNum > 0 ? currentNum.toString().padStart(3, '0') : "无";

            document.getElementById('total-waiting').innerText = queue.length;

            

            // 更新排队列表

            const listEl = document.getElementById('queue-list');

            listEl.innerHTML = queue.slice(0, 5).map(n => `<li><span>号码: ${n.toString().padStart(3, '0')}</span> <span class="status-tag">等待中</span></li>`).join('');

            

            // 更新学生端排队信息

            if(myTicket) {

                let index = queue.indexOf(myTicket);

                document.getElementById('wait-info').innerText = index >= 0 ? `前面还有 ${index} 人在等待` : "正在叫号或已过号";

            }

        }


        // 语音播报函数

        function speak(text) {

            const msg = new SpeechSynthesisUtterance(text);

            msg.lang = 'zh-CN';

            window.speechSynthesis.speak(msg);

        }

    </script>
